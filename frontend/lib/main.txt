import 'dart:convert';
import 'package:flutter/foundation.dart' show kIsWeb;
import 'package:flutter/material.dart';
import 'package:http/http.dart' as http;

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'SRRP',
      theme: ThemeData(primarySwatch: Colors.teal, useMaterial3: true),
      home: const MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  const MyHomePage({super.key});

  @override
  State<MyHomePage> createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  List<dynamic> _resources = [];
  bool _isLoading = true;

  final _nameController = TextEditingController();
  final _typeController = TextEditingController();
  final _capacityController = TextEditingController();

  final String _apiUrl = kIsWeb
      ? 'http://192.168.1.119:8000' // Web için
      : 'http://127.0.0.1:8000'; // Android Emülatör için

  @override
  void initState() {
    super.initState();
    _fetchResources();
  }

  Future<void> _fetchResources() async {
    setState(() {
      _isLoading = true;
    });
    try {
      final response = await http.get(Uri.parse('$_apiUrl/api/resources'));
      if (response.statusCode == 200) {
        setState(() {
          _resources = jsonDecode(utf8.decode(response.bodyBytes));
          _isLoading = false;
        });
      } else {
        print("Listeleme Hatası: ${response.statusCode}");
        setState(() {
          _isLoading = false;
        });
      }
    } catch (e) {
      print("Listeleme Kritik Hata: $e");
      setState(() {
        _isLoading = false;
      });
    }
  }

  Future<void> _addResource() async {
    if (_nameController.text.isEmpty ||
        _typeController.text.isEmpty ||
        _capacityController.text.isEmpty) {
      print("Form alanları boş.");
      return;
    }
    try {
      final response = await http.post(
        Uri.parse('$_apiUrl/api/resources'),
        headers: {'Content-Type': 'application/json; charset=UTF-8'},
        body: jsonEncode({
          'name': _nameController.text,
          'type': _typeController.text,
          'capacity_mw': double.tryParse(_capacityController.text) ?? 0.0,
        }),
      );

      if (response.statusCode == 200) {
        _nameController.clear();
        _typeController.clear();
        _capacityController.clear();
        _fetchResources(); // Listeyi yenile
      } else {
        print("Ekleme Hatası: ${response.statusCode} - ${response.body}");
      }
    } catch (e) {
      print("Ekleme Kritik Hata: $e");
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Yenilenebilir Kaynak Planlayıcı')),
      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(
              'Yeni Kaynak Ekle',
              style: Theme.of(context).textTheme.headlineSmall,
            ),
            TextField(
              controller: _nameController,
              decoration: const InputDecoration(labelText: 'Kaynak Adı'),
            ),
            TextField(
              controller: _typeController,
              decoration: const InputDecoration(
                labelText: 'Türü (Örn: Güneş Paneli)',
              ),
            ),
            TextField(
              controller: _capacityController,
              decoration: const InputDecoration(labelText: 'Kapasite (MW)'),
              keyboardType: TextInputType.number,
            ),
            const SizedBox(height: 12),
            ElevatedButton(
              onPressed: _addResource,
              child: const Text('Yeni Kaynak Ekle'),
            ),
            const Divider(height: 32),
            Text(
              'Mevcut Kaynaklar',
              style: Theme.of(context).textTheme.headlineSmall,
            ),
            Expanded(
              child: _isLoading
                  ? const Center(child: CircularProgressIndicator())
                  : ListView.builder(
                      itemCount: _resources.length,
                      itemBuilder: (context, index) {
                        final resource = _resources[index];
                        return Card(
                          margin: const EdgeInsets.symmetric(vertical: 8),
                          child: ListTile(
                            title: Text(resource['name']),
                            subtitle: Text(resource['type']),
                            trailing: Text('${resource['capacity_mw']} MW'),
                          ),
                        );
                      },
                    ),
            ),
          ],
        ),
      ),
    );
  }
}
